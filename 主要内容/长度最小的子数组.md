```java
class Solution {
    public int minSubArrayLen(int target, int[] nums) {
        int left = 0;
        int right = 0;
        int sum = 0;
        int result = Integer.MAX_VALUE;
        while(right<nums.length){
            sum += nums[right];
            while(sum>=target){
                result =Math.min(right-left+1,result);
                sum-=nums[left];
                left++;
            }
            right++;
        }
        return result = result == Integer.MAX_VALUE? 0:result;
    }
}
```
